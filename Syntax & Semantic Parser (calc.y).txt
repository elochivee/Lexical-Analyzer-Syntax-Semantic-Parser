%{
#include <stdio.h>
#include <stdlib.h>


int yylex();
void yyerror(const char *s);
%}

%token NUMBER
%left '+' '-'
%left '*' '/'
%left '(' ')'


%%

expr:
        expr '+' expr { $$ = $1 + $3;    printf("Result: %d\n", $$); }
        | expr '-' expr { $$ = $1 - $3;  printf("Result: %d\n", $$); }
        | expr '*' expr { $$ = $1 * $3;  printf("Result: %d\n", $$); }
        | expr '/' expr { $$ = $1 / $3;  printf("Result: %d\n", $$); }
        | '(' expr ')' { $$ = $2; }
        | NUMBER { $$ = $1; }
        ;

%%

int main() {
        printf("Enter an expression:\n");
        yyparse();
        return 0;
}

void yyerror(const char *s) {
        fprintf(stderr, "Error: %s\n", s);
}
