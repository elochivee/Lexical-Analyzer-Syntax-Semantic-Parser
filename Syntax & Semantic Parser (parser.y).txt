{

#include <stdlib.h>

%token NUMBER END
%left '+' '-'
%left '*' '/'
%right UMINUS

%%

input:
    /* empty */
    | input line
    ;
line:
    expr END { printf("Result: %d\n", $1); }
    ;

expr:
    NUMBER            { $$ = $1; }
    | expr '+' expr   { $$ = $1 + $3; }
    | expr '-' expr   { $$ = $1 - $3; }
    | expr '*' expr   { $$ = $1 * $3; }
    | expr '/' expr   { if ($3) $$ = $1 / $3; else { printf("Error: Division by>
    | '-' expr %prec UMINUS { $$ = -$2; }
    | '(' expr ')'    { $$ = $2; }
    ;

%%
int main() {
    printf("Enter an arithmetic expression:\n");
    yyparse();
    return 0;
}
